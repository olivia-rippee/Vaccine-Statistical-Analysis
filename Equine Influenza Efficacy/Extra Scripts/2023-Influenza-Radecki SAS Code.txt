Title1 'Duration of Disease - Cases';

data duration1;
	set cases1;

	DPC1 = substrn(_NAME_, 4, 2);
	if DPC1 = '' then DPC1 = '0';

	DPC = DPC1 + 0;

	if case = 0 then delete;

	proc sort;
		by horse_id DPC;
		
	proc print;
	run;

data duration2;
	set duration1;
	by horse_id notsorted;
	if first.horse_id;
	proc print;
	run;

data first;
	set duration2;
	first_day = DPC;
	keep horse_id first_day;
	proc print;
	run;


data duration3;
	set duration1;
	by horse_id notsorted;
	if last.horse_id;
	proc print;
	run;

data last;
	set duration3;
	last_day = DPC;
	keep horse_id trt_group last_day;
	proc print;
	run;

data duration4;
	merge first last;
	by horse_id;
	duration = 1+(last_day - first_day);
	proc sort;
		by horse_id;
	proc print;
	run;

data duration5;
	set cases1;
	if _NAME_ = 'DOC';
	keep horse_id trt_group;
	proc sort;
	by horse_id;
	proc print;
	run;

data duration6;
	merge duration5 duration4;
	by horse_id;
	if duration = . then duration = 0;
	if trt_group = 'control' then trt = 'B'; *control;
	if trt_group = 'Vaccinate' then trt = 'A'; *vaccinate;
	proc print;
	run;


		proc rank data=duration6 out=duration7 ties=mean;
			ranks rank;
			var duration;
		*proc print;
		run;

		proc npar1way data = duration7 wilcoxon;
		class trt;
		var duration;
		run; 

		proc freq data=duration7 noprint;
		tables trt*rank/out=out1 sparse;
		run;
		
		proc sort data=out1;
			by rank trt;
			*proc print;
		run;

		proc freq data=out1;
		weight count;
		tables rank*trt/measures norow nocol;
		test smdrc; 
		output out=zout smdrc ;
		run;

	by halter_id trt_group;
	var DOC
		DOC1
		DOC2
		DOC3
		DOC4
		DOC5
		DOC6
		DOC7
		DOC8
		DOC9
		DOC10
		DOC11
		DOC12
		DOC13
		DOC14;
	proc sort;
	by halter_id trt_group _NAME_;
	proc print;
	run;

data temp2;
	set temp1;
	temp = col1 + 0;

	if temp ge 102.5 then pyrexia = 1;
	if temp < 102.5 then pyrexia = 0;

	proc print;
	run;

proc means data = temp2 noprint;
	by halter_id trt_group;
	var pyrexia;
	output out = temp3 sum(pyrexia) = days_pyrexic;
	proc print;
	run;

data temp4;
	set temp3;
	if days_pyrexic > 0 then pyr = 1;
	if days_pyrexic = 0 then pyr = 0;
	proc print;
	run;

proc freq data = temp4;
	tables pyr*trt_group;
	exact fisher;
	run;
